#include <Servo.h>
Servo myServo;
int rainSensor = A0;
void setup() {
    myServo.attach(9);
    pinMode(rainSensor, INPUT);
}
void loop() {
    int rainValue = analogRead(rainSensor);
    if (rainValue < 500) { 
        myServo.write(90);
    } else {
        myServo.write(0);
    }
    delay(1000);
}
